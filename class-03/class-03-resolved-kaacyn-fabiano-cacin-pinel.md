## 3. Liste todos Pokemons com a altura menor ou igual que 0.5 E do tipo grama;

```
var query = { name: "Bulbasaur" } ;
var p = db.pokemons.findOne(query)
p.type = "grama";
p.height = 0.4;
db.pokemons.save(p);

var query = { $and: [ {height: $lte : {0.5}}, {type: 'grama']} } ;
db.pokemons.find(query)
